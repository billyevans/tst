1//! Ternary search trie (TST) container.
2
3/// - Create a `TSTMap` containing a given list of elements:
4///
5/// # Examples
6///
7/// ```
8/// #[macro_use]
9/// extern crate tst;
10/// # fn main() {
11/// let m = tstmap!{
12///     "b" => 2, "a" => -1, "c" => 3,
13/// };
14///
15/// assert_eq!(3, m.len());
16/// assert_eq!(m["a"], -1);
17/// assert_eq!(m["b"], 2);
18/// assert_eq!(m["c"], 3);
19/// # }
20/// ```
21#[macro_export]
22macro_rules! tstmap {
23    () => {{
24        $crate::TSTMap::new()
25    }};
26    // trailing comma case
27    ($($key:expr_2021 => $value:expr_2021,)+) => (tstmap!($($key => $value),+));
28    ($( $key: expr_2021 => $val: expr_2021 ),*) => {{
29        let mut m = $crate::TSTMap::new();
30        $(
31            m.insert($key, $val);
32        )*
33        m
34    }};
35}

37/// - Create a `TSTSet` containing a given list of elements:
38///
39/// # Examples
40///
41/// ```
42/// #[macro_use]
43/// extern crate tst;
44/// # fn main() {
45/// let s = tstset!{
46///     "bcb", "bca", "bcd"
47/// };
48///
49/// assert_eq!(3, s.len());
50/// assert!(!s.contains("b"));
51/// assert!(!s.contains("bc"));
52/// assert!(s.contains("bca"));
53/// assert!(s.contains("bcb"));
54/// assert!(s.contains("bcd"));
55/// # }
56/// ```
57#[macro_export]
58macro_rules! tstset {
59    () => {{
60        $crate::TSTSet::new()
61    }};
62    // trailing comma case
63    ($($key:expr_2021,)+) => (tstset!($($key),+));
64    ($($key: expr_2021),*) => {{
65        let mut s = $crate::TSTSet::new();
66        $(
67            s.insert($key);
68        )*
69        s
70    }};
71}
